# 枚举
* 定义：一种自定义数据类型，允许定义一组命名的常量表示某个有限集合的可能值
* 如，一周几天，一月几天，扑克牌花色等固定事物
* 可增加可读性
````
#include<stdio.h>

enum Day
{
	Mon,
	Tues,
	Wed,
	Thur,
	Fri,
	Sat,
	Sun
};


enum color
{
	RED,
	YELLOW,
	GREEN
};
int main()
{
	printf("%d\n", Mon);//0
	printf("%d\n", Tues);//1
	printf("%d\n", Wed);//2

	return 0;
````

* 即可赋值，无赋值数往后顺延
````
enum Traffic_Light
{
	RED,
	GREEN,
	YELLOW
};
int main()
{
	//printf("%d\n", Mon);//1
	//printf("%d\n", Tues);//2
	//printf("%d\n", Wed);//3

	/*enum color clr = GREEN;
	printf("%d\n", clr);*/


	enum Traffic_Light light = RED;
	for (int i = 0; i < 3; i++)
	{
		switch (light)
		{
		    case RED:
				printf("红灯停\n");
				light = GREEN;
				break;
			case GREEN:
				printf("绿灯行\n");
				light = YELLOW;
				break;
			case YELLOW:
				printf("黄灯减速\n");
				light = RED;
				break;
		}

	}


	return 0;
````
